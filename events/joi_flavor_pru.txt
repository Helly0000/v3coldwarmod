joi_flavor_pru.5 = {#守望莱茵
	type = country_event
	placement = root
	title = joi_flavor_pru.5.t
	desc = joi_flavor_pru.5.d
	flavor = joi_flavor_pru.5.f

	duration = 3

	event_image = {
		video = "europenorthamerica_before_the_battle"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/bicorne_hat.dds"

	trigger = {
		any_primary_culture = {
			OR = { 
				cu:south_german = this 
				cu:north_german = this
			}
		}
		NOT = {
			c:AUS = THIS
			}
		c:FRA = {
			has_modifier = france_natural_borders_pursuit
		}
	}

	immediate = {
		c:FRA = {
            save_scope_as = FRA_scope
        }
	}

	option = {

		trigger = {
			NOT = {
			c:PRU = THIS
			c:NGF = THIS
			c:GER = THIS
			}
		}
		name = joi_flavor_pru.5.a
		if = {
			limit = {
			exists = c:PRU
			}
			set_owes_obligation_to = {
				country = c:PRU
				setting = yes
				}
			}
		if = {
			limit = {
			exists = c:NGF
			}
			set_owes_obligation_to = {
				country = c:NGF
				setting = yes
				}
			}
		if = {
			limit = {
			exists = c:GER
			}
			set_owes_obligation_to = {
				country = c:GER
				setting = yes
				}
			}
		add_modifier = {
				name = defend_rhine
				months = 30
		}
		every_scope_state = {
			limit = {
				OR = {
					state_region = s:STATE_WALLONIA
					state_region = s:STATE_FLANDERS
					state_region = s:STATE_RHINELAND
					state_region = s:STATE_NORTH_RHINE
					state_region = s:STATE_SAVOY
				}
			}
			add_modifier = {
				name = rally_against_france
				months = 120
			}
		}
	}

	option = {

		trigger = {
			OR = {
			c:PRU = THIS
			c:NGF = THIS
			c:GER = THIS
			}
		}
		name = joi_flavor_pru.5.b
		add_modifier = {
				name = defender_of_rhine
				months = 30
		}
		every_scope_state = {
			limit = {
				OR = {
					state_region = s:STATE_WALLONIA
					state_region = s:STATE_FLANDERS
					state_region = s:STATE_RHINELAND
					state_region = s:STATE_NORTH_RHINE
					state_region = s:STATE_SAVOY
				}
			}
			add_modifier = {
				name = rally_against_france
				months = 120
			}
		}
	}
}
