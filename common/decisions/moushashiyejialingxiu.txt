murder_of_an_industrialist_leader = { #谋杀实业家领袖
	is_shown = {
			has_modifier = activate_resolution_to_murder_interest_group_leaders #有谋杀利益集团领袖修正
	}
	possible = {	
			NOT = { has_modifier = murder_funding5 } #没有谋杀经费修正	
	}
	when_taken = {
		create_character = {
			ig_leader = yes
			interest_group = ig_industrialists #实业家
			}
			add_modifier = {
			name = activate_resolution_to_murder_interest_group_leaders #谋杀利益集团领袖
		}
			add_modifier = {
			name = murder_funding5 #谋杀经费
			years = 5	
		}
	}
}
