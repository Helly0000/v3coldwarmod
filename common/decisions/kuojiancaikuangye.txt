expanding_the_mining_industry = { #扩建采矿业
	is_shown = {
			is_ai = no
		NOT = { 
			has_modifier = expanding_the_mining_industry
		}
	}
	possible = {
		produced_bureaucracy > 100	
			any_scope_state = {
			any_scope_building = {
				is_building_type = building_iron_mine
				level >= 11
			}
		}	
	}
	
	when_taken = {
		add_modifier = {
			name = expanding_the_mining_industry
		}
	}
}
